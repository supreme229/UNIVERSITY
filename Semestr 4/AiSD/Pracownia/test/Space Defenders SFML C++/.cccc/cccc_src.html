           if (  selectIndex == 3 )
           {

           }
           }
           if (  eventSF.key.code == sf::Keyboard::Left && selectIndex == 1 )
           {
           this->levelPrev(   );
           }
           if (  eventSF.key.code == sf::Keyboard::Right && selectIndex == 1 )
           {
           this->levelNext(   );
           }
           if (  selectIndex == 2 )
           {
           this->tutorialShow = true;
           }
           else
